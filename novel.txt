import os

def create_novel_page(title, content, is_premium=False):
    template = f"""
    <!DOCTYPE html>
    <html lang="ar" dir="rtl">
    <head><title>{title}</title></head>
    <body>
        <h1>{title}</h1>
        {"<p>هذا محتوى مدفوع...</p>" if is_premium else f"<p>{content}</p>"}
    </body>
    </html>
    """
    filename = f"{title.replace(' ', '_')}.html"
    with open(filename, "w", encoding="utf-8") as f:
        f.write(template)
    print(f"تم إنشاء الصفحة: {filename}")

# مثال للاستخدام
create_novel_page("الفصل الأول مجانا", "كان يا ما كان في قديم الزمان...")
create_novel_page("الفصل الثاني مدفوع", "هنا الأسرار التي لا تظهر إلا بالدفع", is_premium=True)
